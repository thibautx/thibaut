for i in range(len(headers)):
    fig = plt.figure(figsize=(15, 7))
    header = headers[i]
    ax = fig.add_subplot(111)
    ax.set_title('{} vs {}'.format(header, 'median home value'))
    ax.set_xlabel(headers_descriptions[header])
    ax.set_ylabel('Median value of owner-occupied homes in $1000s')
    ax.scatter(X[header], y['MEDV'])
    
    # fit linear-regression
    reg = linear_model.LinearRegression(fit_intercept=True)
    reg.fit(X[header].reshape(len(X[header]), 1), y['MEDV'].reshape(len(y['MEDV'])), 1)
    regression_x = np.linspace(np.min(X[header]), np.max(X[header]), num=100)
    regression_y = map(lambda x: reg.coef_[0]*x + reg.intercept_, regression_x)
    ax.plot(regression_x, regression_y)
    
plt.show()
